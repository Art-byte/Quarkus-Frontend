import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { ThemePalette } from '@angular/material/core';
import { Router } from '@angular/router';
import { User } from 'src/app/shared/models/user';
import { UserService } from 'src/app/shared/services/user.service';
import { Constants } from 'src/app/shared/utils/actionconstants';
import { updatedConfirmation, updatedProfile } from 'src/app/shared/utils/alerts';
import Swal from 'sweetalert2';
import { FileValidator } from 'ngx-material-file-input';
import { environment } from 'src/environments/environment';
import { MatDialog } from '@angular/material/dialog';
import { ChangePhotoComponent } from './change-photo/change-photo.component';


@Component({
  selector: 'app-profile',
  templateUrl: './profile.component.html',
  styleUrls: ['./profile.component.css'],
})
export class ProfileComponent implements OnInit {
  user: User = new User();
  form: FormGroup;
  formPassword: FormGroup;
  userProfile = localStorage.getItem('username');
  _id: string;

  color: ThemePalette = 'accent';
  checked = false;
  showDetails: boolean = false;
  urlEndPoint = environment.apiUrl;

  constructor(
    private router: Router,
    private userService: UserService,
    private formBuilder: FormBuilder,
    public dialog: MatDialog
  ) {
    //formulario de informacion del perfil
    this.form = this.formBuilder.group({
      name: [''],
      lastName: [''],
      username: [''],
      email: [''],
      phone: [''],
    });

    //formulario de cambio de password
    this.formPassword = this.formBuilder.group({
      newPassword: ['', [Validators.required, Validators.pattern(new RegExp(Constants.pattertPassword))]],
      confirmPassword: ['', Validators.required],
    }, {
      validators: this.IsEquals('newPassword', 'confirmPassword')
    });


  }

  onStrengthChangeds(strength: number) {
  }

    // Gets para devolver los valores de los campos del fomulario de cambiar la contrase√±a
    get newPassword() {
      return this.formPassword.get('newPassword');
    }
    get confirmPassword() {
      return this.formPassword.get('confirmPassword');
    }

    IsEquals(pNewPassword, pConfirmPassword) {
      return (formGroup: FormGroup) => {
        const pass1Control = formGroup.get(pNewPassword);
        const pass2Control = formGroup.get(pConfirmPassword);
        if (pass1Control.value === pass2Control.value) {
          pass2Control.setErrors(null);
        } else {
          pass2Control.setErrors({ noEsIgual: true });
        }
      };
    }
  

  /**
   * Al iniciar el componente, nos carga
   * la informacion del usuario que se encuentra logeado
   */
  ngOnInit(): void {
    this.getUserInfo();
  }

  updateUserPassword(){
    const username = localStorage.getItem('username');
    this.userService.getUsername(username).subscribe(resp => {
      this.userService.updatePassword(resp.id, this.newPassword.value, this.confirmPassword.value, resp).subscribe(data => {
        updatedConfirmation()
        this.formPassword.reset();
        this.router.navigate(['dashboard']);
      });
    });
  }



  updateProfile() {
    const username = localStorage.getItem('username');
    this.userService.getUsername(username).subscribe(user => {
      this.userService.updateProfile(user.id, this.user).subscribe((data) => {
        updatedProfile() //Alerta
      });
    });
  }

  //alerta de informes
  updateProfileMessageConfirmation = () => {
    Swal.fire({
      icon: 'info',
      title: 'Actualizacion de perfil',
      text: 'Esta apunto de actualizar la informacion de su perfil',
      confirmButtonColor: '#3f51b5',
      cancelButtonColor: '#3f51b5',
    }).then((result) => {
      if (result.isConfirmed) {
       this.updateProfile();
       this.getUserInfo();
      }
    });
  };



  getUserInfo(){
    this.userService.getUsername(this.userProfile).subscribe((formData) => {
      this.user = formData;
    }),
      (error) => {
        console.log(error);
      };
  }


  openDialogPerfil(): void {
    const dialogRef = this.dialog.open(ChangePhotoComponent, {
      maxWidth: '500px',
      maxHeight: '600px',
      data: this.user,
      id: 'dialog-foto-perfil',
      autoFocus: false
    });
    dialogRef.afterClosed().subscribe(result => {
      this.user.profilePicture = result.foto;
      // if (this.usuario.id === this.authService.usuario.id) {
      //   this.topnavService.notificarFoto.emit(result.foto);
      // }
    });
  }

}
