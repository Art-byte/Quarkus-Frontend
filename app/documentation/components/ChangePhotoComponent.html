import { Component, Inject, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { Router } from '@angular/router';
import { FileValidator } from 'ngx-material-file-input';
import { UserService } from 'src/app/shared/services/user.service';
import { changeProdfilePhotoConfirmation, quitarBackground } from 'src/app/shared/utils/alerts';

@Component({
  selector: 'app-change-photo',
  templateUrl: './change-photo.component.html',
  styleUrls: ['./change-photo.component.scss']
})
export class ChangePhotoComponent implements OnInit {

  extensions = ['image/png', 'image/jpg', 'image/x-icon', 'image/jpeg'];
  photoProfile: FormGroup;

  constructor(    
    private router: Router,
    private userService: UserService,
    private formBuilder: FormBuilder,
    public dialogRef: MatDialogRef<ChangePhotoComponent>,
    @Inject(MAT_DIALOG_DATA) public data: any
    ) { 
          //Cambio de foto de perfil 
    this.photoProfile = this.formBuilder.group({
      profilePicture: [null, [Validators.required, FileValidator.maxContentSize(10485760)]]
    }, {
      validators: this.validatePhoto('profilePicture')
    });
    }

  ngOnInit(): void {
    this.dialogRef.disableClose = true;
    quitarBackground('dialog-foto-perfil');
  }

    //Controles sobre la foto de perfil 
    get profilePicture() { return this.photoProfile.get('profilePicture'); }

    validatePhoto(archivo) {
      return (formGroup: FormGroup) => {
        const archivo1 = formGroup.get(archivo);
        if (archivo1.value !== null) {
          this.extensions.includes(archivo1.value.files[0].type) ?
            archivo1.setErrors(null) :
            archivo1.setErrors({ fileInvalid: true });
        }
      };
    }

    close() {
      this.dialogRef.close(this.data);
    }
  
      //Revisar esta actividad con referencia al back
  guardar() {
    this.userService.updatePhoto(this.data.id, this.profilePicture.value.files[0]).subscribe(data => {
      changeProdfilePhotoConfirmation();
    });
  }

}
